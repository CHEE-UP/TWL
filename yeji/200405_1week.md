# class Solution:
def moveZeroes(nums) -> None:
    countZeros = nums.count(0)
    for _ in range(countZeros):
        nums.remove(0)
        nums.append(0) 
    # return nums
    print(nums)
    
moveZeroes([1,3,12,4,0,1,0])

- HappyNumber

```python

class Solution:
    def isHappy(self, n: int) -> bool:
        while n != 1:
            n = sum([int(i)**2 for i in str(n)])
            if n == 4:
                return False
        return True

```
