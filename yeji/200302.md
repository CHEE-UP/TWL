```python
    answer = True
```

```python
# 입출력예에서 A가 오름차순이고, B가 내림차순일때
# 최솟값이 나온다는 규칙을 찾아냄 ㅎㅎ
def solution(A,B):
    answer = 0
    answers = []
    sA = sorted(A)
    bB = sorted(B, reverse=True)
    for i1, i2 in zip(sA, bB):
        answers.append(i1*i2)
    for i in answers:
        answer = answer + i
    return answer

# 다른사람풀이
```

```python
def solution(s):
    answer = []
    ss = s.split(" ")
    for i in ss:
        if i == min(ss):
            answers = []
            answers.append(i)
        elif i == max(ss):
            answers.append(i)
        elif min(ss) == max(ss):
            answers.append(i)
    return " ".join(answers)

def solution(s):
    answer = []
    answers = []
    ss = s.split(" ")
    for i in ss:
        answers.append(int(i))
    answer.append(min(answers))
    answer.append(max(answers))
    answerss = [str(i) for i in answer]

    return " ".join(answerss)
```
